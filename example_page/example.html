גלריה

ביוגרפיה

נתנאל אלקים נולד בישראל בשנת 1982. הוא גדל בעיירה קטנה בגליל, מוקף בטבע ובנוף ההררי. כבר בילדותו גילה כישרון יוצא דופן בתחום המוזיקה וניגן במספר כלי נגינה. עם השנים, פיתח אהבה מיוחדת לגיטרה והחל ליצור מוזיקה מקורית שהושפעה מהשילוב בין מסורת למודרניות. עם הזמן, נתנאל ייסד סטודיו עצמאי להפקת מוזיקה שבו עבד עם אמנים מכל רחבי הארץ. הוא התפרסם בזכות יכולתו ללכוד רגעים ייחודיים ורגשות דרך המוזיקה, והפך לדמות בולטת בתעשיית המוזיקה הישראלית. עבודתו שילבה יצירתיות טכנית עם רגישות אמנותית, והוא היה למנטור למוזיקאים צעירים רבים. נתנאל זכור לא רק בזכות המוזיקה שיצר, אלא גם בזכות טוב ליבו ונכונותו לעזור לכל אדם. הוא הותיר אחריו מורשת של אהבה למוזיקה, משפחה ותרומה לקהילה. זכרו של נתנאל יישאר בליבם של משפחתו, חבריו וכל מי שהכיר אותו. המוזיקה שלו תמשיך להדהד ולגעת בלבבות רבים. יהי זכרו ברוך.

תהילים פרק קיט

תהילים קיט - אות נ

נֵר-לְרַגְלִי דְבָרֶךָ, וְאוֹר לִנְתִיבָתִי.
נִשְׁבַּעְתִּי וָאֲקַיֵּמָה, לִשְׁמֹר מִשְׁפְּטֵי צִדְקֶךָ.
נַעֲנֵיתִי עַד-מְאֹד; יְהוָה, חַיֵּנִי כִּדְבָרֶךָ.
נְדוּבוֹת פִּי, רְצֵה-נָא יְהוָה; וּמִשְׁפָּטֶיךָ לַמְּדֵנִי.
נַפְשִׁי בְכַפִּי תָמִיד, וְתוֹרָתְךָ לֹא שָׁכָחְתִּי.
נָתְנוּ רְשָׁעִים פַּח לִי, וּמִפִּקּוּדֶיךָ לֹא תָעִיתִי.
נָחַלְתִּי עֵדְוֹתֶיךָ לְעוֹלָם: כִּי-שְׂשׂוֹן לִבִּי הֵמָּה.
נָטִיתִי לִבִּי, לַעֲשׂוֹת חֻקֶּיךָ; לְעוֹלָם, עֵקֶב.

תהילים קיט - אות ת

תִּקְרַב רִנָּתִי, לְפָנֶיךָ יְהוָה; כִּדְבָרְךָ הֲבִינֵנִי.
תָּבוֹא תְּחִנָּתִי, לְפָנֶיךָ; כְּאִמְרָתְךָ הַצִּילֵנִי.
תַּבַּעְנָה שְׂפָתַי, תְּהִלָּה: כִּי תְלַמְּדֵנִי, חֻקֶּיךָ.
תַּעַן לְשׁוֹנִי, אִמְרָתֶךָ: כִּי כָּל-מִצְו‍ֹתֶיךָ צֶדֶק.
תְּהִי יָדְךָ, לְעָזְרֵנִי: כִּי פִקּוּדֶיךָ בָחָרְתִּי.
תָּאַבְתִּי, לִישׁוּעָתְךָ יְהוָה; וְתוֹרָתְךָ, שַׁעֲשֻׁעָי.
תְּחִי נַפְשִׁי, וּתְהַלְלֶךָּ; וּמִשְׁפָּטֶיךָ יַעֲזְרֻנִי.
תָּעִיתִי, כְּשֶׂה אֹבֵד; בַּקֵּשׁ עַבְדֶּךָ-- כִּי מִצְו‍ֹתֶיךָ, לֹא שָׁכָחְתִּי.

תהילים קיט - אות א

אַשְׁרֵי תְמִימֵי-דָרֶךְ-- הַהֹלְכִים, בְּתוֹרַת יְהוָה.
אַשְׁרֵי, נֹצְרֵי עֵדֹתָיו; בְּכָל-לֵב יִדְרְשׁוּהוּ.
אַף לֹא-פָעֲלוּ עַוְלָה; בִּדְרָכָיו הָלָכוּ.
אַתָּה צִוִּיתָה פִקֻּדֶיךָ-- לִשְׁמֹר מְאֹד.
אַחֲלַי, יִכֹּנוּ דְרָכָי; לִשְׁמֹר חֻקֶּיךָ.
אָז לֹא-אֵבוֹשׁ; בְּהַבִּיטִי, אֶל-כָּל-מִצְו‍ֹתֶיךָ.
אוֹדְךָ, בְּיֹשֶׁר לֵבָב-- בְּלָמְדִי מִשְׁפְּטֵי צִדְקֶךָ.
אֶת-חֻקֶּיךָ, אֶשְׁמֹר; אַל-תַּעַזְבֵנִי עַד-מְאֹד.

תהילים קיט - אות ל

לְעוֹלָם, יְהוָה-- דְּבָרְךָ, נִצָּב בַּשָּׁמָיִם.
לְדוֹר וָדוֹר, אֱמוּנָתֶךָ; כּוֹנַנְתָּ אֶרֶץ, וַתַּעֲמֹד.
לְמִשְׁפָּטֶיךָ, עָמְדוּ הַיּוֹם: כִּי הַכֹּל, עֲבָדֶיךָ.
לֻלֵי תוֹרָתְךָ שַׁעֲשֻׁעָי-- אָז, אָבַדְתִּי בְעָנְיִי.
לְעוֹלָם, לֹא-אֶשְׁכַּח פִּקּוּדֶיךָ: כִּי-בָם חִיִּיתָנִי.
לְךָ-אֲנִי, הוֹשִׁיעֵנִי: כִּי פִקּוּדֶיךָ, דָּרָשְׁתִּי.
לִי קִוּוּ, רְשָׁעִים-- לְאַבְּדֵנִי; עֵדֹתֶיךָ, אֶתְבּוֹנָן.

תהילים קיט - אות ב

בָּרוּךְ אַתָּה יְהוָה; לַמְּדֵנִי חֻקֶּיךָ.
בְּשִׂפְתֵי, סִפַּרְתִּי; כָּל-מִשְׁפְּטֵי פִיךָ.
בְּדֶרֶךְ עֵדוֹתֶיךָ, שָׂשׂתִּי; כְּעַל כָּל-חוֹשֶׁק.
בְּפִקּוּדֶיךָ, אָשִׂיחָה; וְאַבִּיטָה אֹרְחוֹתֶיךָ.
בְּחֻקּוֹתֶיךָ, אֶשְׁתַּעֲשָׁע; לֹא אֶשְׁכַּח דְּבָרֶךָ.

תהילים קיט - אות מ

מַה-אָהַבְתִּי, תוֹרָתֶךָ: כָּל-הַיּוֹם, הִיא שִׂיחָתִי.
מֵאוֹיְבַי, תְּחַכְּמֵנִי מִצְו‍ֹתֶךָ: כִּי לְעוֹלָם הִיא-לִי.
מִכָּל-מְלַמְּדַי, הִשְׂכַּלְתִּי: כִּי עֵדְוֹתֶיךָ, שִׂיחָה לִי.
מִזְּקֵנִים, אֶתְבּוֹנָן: כִּי פִקּוּדֶיךָ נָצָרְתִּי.
מִכָּל-דֶּרֶךְ רָע, כָּלִאתִי רַגְלָי: לְמַעַן, אֶשְׁמֹר דְּבָרֶךָ.
מִמִּשְׁפָּטֶיךָ, לֹא-סָרְתִּי: כִּי-אַתָּה הוֹרֵתָנִי.
מַה-נִּמְלְצוּ לְחִכִּי, אִמְרָתֶךָ: מִדְּבַשׁ לְפִי.

תהילים קיט - אות ש

שָׂרִים, רְדָפוּנִי חִנָּם; וּמִדְּבָרְךָ, פָּחַד לִבִּי.
שָׂשׂ אָנֹכִי, עַל-אִמְרָתֶךָ; כְּמוֹצֵא, שָׁלָל רָב.
שֶׁקֶר, שָׂנֵאתִי וַתּוֹעֲבָה; תּוֹרָתְךָ, אָהָבְתִּי.
שֶׁבַע בַּיּוֹם, הִלַּלְתִּיךָ: עַל מִשְׁפְּטֵי צִדְקֶךָ.
שָׁלוֹם רָב, לְאֹהֲבֵי תוֹרָתֶךָ; וְאֵין-לָמוֹ מִכְשׁוֹל.
שִׂבַּרְתִּי, לִישׁוּעָתְךָ יְהוָה; וּמִצְו‍ֹתֶיךָ עָשִׂיתִי.
שָׁמְרָה נַפְשִׁי, עֵדְוֹתֶיךָ; וָאֹהֲבֵם מְאֹד.

תהילים קיט - אות ה

הוֹרֵנִי יְהוָה, דֶּרֶךְ חֻקֶּיךָ; וְאֶצְּרֶנָּה עֵקֶב.
הֲבִינֵנִי, וְאֶצְּרָה תוֹרָתֶךָ; וְאֶשְׁמְרֶנָּה בְכָל-לֵב.
הַדְרִיכֵנִי, בִּנְתִיב מִצְו‍ֹתֶיךָ: כִּי-בוֹ חָפָצְתִּי.
הַט-לִבִּי, אֶל-עֵדְוֹתֶיךָ; וְאַל אֶל-בָּצַע.
הָעֲבֵר עֵינַי מֵרְאוֹת שָׁוְא; בִּדְרָכֶךָ חַיֵּנִי.
הָקֵם לְעַבְדְּךָ, אִמְרָתֶךָ: אֲשֶׁר, לְיִרְאָתֶךָ.
הַעֲבֵר חֶרְפָּתִי, אֲשֶׁר יָגֹרְתִּי: כִּי מִשְׁפָּטֶיךָ טוֹבִים.
הִנֵּה תָאַבְתִּי לְפִקּוּדֶיךָ; בְּצִדְקָתְךָ חַיֵּנִי.
